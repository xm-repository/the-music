#sql完整语句
#用户表
create table user(
                     user_id int primary key auto_increment,
                     user_name varchar(20) default '未命名',
                     user_sex varchar(4) default '未知',
                     user_phone char(11) not null,
                     user_picture varchar(200)
);
#登陆表
create table login(
                      login_id int primary key auto_increment,
                      login_name varchar(8) not null,
                      user_id int,
                      password varchar(16) not null,
                      login_type varchar(6) not null,
                      foreign key(user_id) references user(user_id)
);
#我喜欢信息表
create table ilike_info(
                           id int primary key auto_increment,
                           user_id int,
                           foreign key (user_id) references user(user_id)
);

#歌单
create table song_sheet(
                           sheet_id int primary key auto_increment,
                           sheet_name varchar(20) not null,
                           user_id int,
                           language varchar(50),
                           style varchar(50),
                           mood varchar(50)
);
#歌手
create table singer(
                       singer_id int primary key auto_increment,
                       singer_name varchar(20) not null,
                       singer_sex char(2),
                       singer_area varchar(20) not null,
                       singer_birth date,
                       singer_country varchar(50),
                       singer_height int,
                       singer_weight int,
                       blood_type varchar(10),
                       constellation varchar(6),
                       synopsis varchar(1000),
                       singer_picture varchar(200) not null,
                       fans int not null,
                       singer_name_initials char(2) not null
);
#专辑
create table album(
                      album_id int primary key auto_increment,
                      album_name varchar(20) not null,
                      singer_id int not null,
                      publish_date date not null,
                      foreign key (singer_id) references  singer(singer_id)
);
#歌曲
create table song(
                     song_id int primary key auto_increment,
                     song_name varchar(20) not null,
                     singer_id int,
                     publish_date date not null,
                     song_click int not null,
                     song_src varchar(200) not null,
                     song_length  int,
                     picture_src varchar(200),
                     song_language varchar(20),
                     album_id int,
                     lylic_src varchar(200),
                     foreign key(singer_id) references singer(singer_id)
);
#歌单歌曲关系表
create table sheet_song_relation(
                                     id int primary key auto_increment,
                                     song_sheet_id int,
                                     song_id int,
                                     foreign key(song_sheet_id) references song_sheet(sheet_id),
                                     foreign key(song_id) references song(song_id)
);
#mv
create table mv(
                   mv_id int primary key auto_increment,
                   mv_name varchar(20) not null,
                   mv_picture varchar(200) not null,
                   mv_src varchar(200) not null,
                   mv_click int not null,
                   song_id int,
                   mv_length int not null,
                   foreign key(song_id) references song(song_id)
);
#歌曲我喜欢关系表
create table song_ilike_relation(
                                     id int primary key auto_increment,
                                     song_id int,
                                     ilike_id int,
                                     foreign key(song_id) references song(song_id),
                                     foreign key(ilike_id) references ilike_info(id)
);
#歌单我喜欢关系表
create table sheet_ilike_relation(
                                     id int primary key auto_increment,
                                     song_sheet_id int,
                                     ilike_id int,
                                     foreign key(song_sheet_id) references song_sheet(sheet_id),
                                     foreign key(ilike_id) references ilike_info(id)
);
#专辑我喜欢关系表
create table album_ilike_relation(
                                     id int primary key auto_increment,
                                     album_id int,
                                     ilike_id int,
                                     foreign key(album_id) references album(album_id),
                                     foreign key(ilike_id) references ilike_info(id)
);
#mv我喜欢关系表
create table mv_ilike_relation(
                                     id int primary key auto_increment,
                                     mv_id int,
                                     ilike_id int,
                                     foreign key(mv_id) references mv(mv_id),
                                     foreign key(ilike_id) references ilike_info(id)
);
#歌曲评论
create table song_comment(
                             scomment_id int primary key auto_increment,
                             song_id int,
                             user_id int,
                             scomment_click int not null,
                             comment_time datetime,
                             #记录是否是回复评论，null为歌曲评论,否则记录为回复评论
                             recomment_id int,
                             content varchar(200) not null,
                             foreign key(song_id) references song(song_id),
                             foreign key(user_id) references user(user_id)
);

#评论点赞表
create table like_comment(
                             id  int primary key auto_increment,
                             user_id int,
                             scomment_id int,
                             foreign key (user_id) references user(user_id),
                             foreign key (scomment_id) references song_comment(scomment_id)
);
